#Formation style that creates groups of ships
formation "Navy Box"
	flippable y
	position 0 174
	#Line that creates the sides of each box going to the right
	line
		start 150 87
		end -150 87
		positions 2
		repeat
			start 750 435
			end 750 435
	#Line that creates the front and back, going to the right
	line
		start 750 435
		end 750 609
		positions 2
		repeat
			start 750 435
			end 750 435
	#Line that creates the sides of each box to the left, with offset
	line
		start -600 522
		end -900 522
		positions 2
		repeat
			start -750 435
			end -750 435
	#Line that creates the front and back, going to the left. Both are offset to avoid flagship
	line
		start -750 435
		end -750 609
		positions 2
		repeat
			start -750 435
			end -750 435
	#Line that holds front and back for the rear line, going to the right
	line
		start 0 870
		end 0 1044
		positions 2
		repeat
			start 750 435
			end 750 435
	
	#Left and right for rear line, to the right
	line
		start -150 957
		end 150 957
		positions 2
		repeat
			start 750 435
			end 750 435
	#Line that holds front and back for the rear line, going to the left and offset
	line
		start -750 1305
		end -750 1479
		positions 2
		repeat
			start -750 435
			end -750 435
	#left and right for rear line, to the left with offset
	line
		start -600 1392
		end -900 1392
		positions 2
		repeat
			start -750 435
			end -750 435
